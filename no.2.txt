import cv2
image = cv2.imread('C:/Users/kavin/Downloads/images.jpg')
if image is None:
    print("Error: Unable to load image.")
else:
    blurred_image = cv2.GaussianBlur(image, (15, 15), 0)  # Kernel size (15, 15) can be adjusted

    cv2.imshow('Original Image', image)
    cv2.imshow('Blurred Image', blurred_image)

    cv2.waitKey(0)
    cv2.destroyAllWindows()
